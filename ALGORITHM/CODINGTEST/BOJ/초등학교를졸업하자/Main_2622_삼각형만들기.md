import java.util.Scanner;

public class Main {
	public static void main(String[] args) throws Exception{
		Scanner sc = new Scanner(System.in);
		int n = sc.nextInt();
		int cnt = 0;
		for(int i = 1; i < n; i++) {
			for(int j = i; j < n; j++) {
				int k = n-(i+j);
				if(j > k) break;
				if(i+j > k) cnt++;
			}
		}
		System.out.println(cnt);
	}
}
